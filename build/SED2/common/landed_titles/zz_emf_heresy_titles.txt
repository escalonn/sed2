d_ismaili = { # The Isma'ili Caliphate
	color = { 60 190 60 }
	color2 = { 220 220 0 }
	capital = 719 # Makkah # Mecca
	creation_requires_capital = no
	dignity = 100 # Counted as having this many more counties than it does
	religion = ismaili
	# Controls a religion
	controls_religion = ismaili
	assimilate = no
	caliphate = yes
	coat_of_arms = {
		data = { 0 0 0 9 1 7 7 }
		religion = ismaili
	}
	gain_effect = {
		if = {
			limit = { d_ismaili = { NOT = { has_title_flag = ismaili_followers } } }
			hidden_effect = {
				character_event = {
					id = emf_heresy.1973
					days = 30
				}
			}
		}
	}
	allow = {
		OR = {
			trait = mirza
			trait = sayyid
			piety = 1000
		}
		trigger_if = {
			limit = {
				NOT = {
					has_alternate_start_parameter = {
						key = religion_names
						value = random
					}
				}
			}
			custom_tooltip = {
				text = ismaili_creation_independent
				hidden_tooltip = {
					in_revolt = no
					OR = {
						independent = yes
						NOT = {
							any_liege = {
								OR = {
									in_revolt = yes
									NOT = { religion = ismaili }
								}
							}
						}
					}
				}
			}
			custom_tooltip = {
				text = not_shia_caliphate_revolt_ongoing
				hidden_tooltip = { NOT = { has_global_flag = ismaili_caliphate_revolt_ongoing } }
			}
			OR = {
				AND = {
					trait = sayyid
					piety = 1000
				}
				trigger_if = {
					limit = { NOT = { has_global_flag = destroyed_caliphates } }
					custom_tooltip = {
						text = controls_mecca_medina
						hidden_tooltip = {
							c_mecca = { owner_under_ROOT = yes }
							c_medina = { owner_under_ROOT = yes }
						}
					}
				}
				custom_tooltip = {
					text = controls_jerusalem_damascus_baghdad
					hidden_tooltip = {
						c_jerusalem = { owner_under_ROOT = yes }
						c_damascus = { owner_under_ROOT = yes }
						c_baghdad = { owner_under_ROOT = yes }
					}
				}
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = THREE_HOLY_SITES_TT
				hidden_tooltip = {
					any_realm_title = {
						count = 3
						is_holy_site = ROOT
					}
				}
			}
		}
		trigger_if = {
			limit = { has_global_flag = destroyed_caliphates }
			custom_tooltip = {
				text = controls_mecca_medina
				hidden_tooltip = {
					c_mecca = { owner_under_ROOT = yes }
					c_medina = { owner_under_ROOT = yes }
				}
			}
		}
	}
	title = CALIPH
	title_female = CALIPHA
	foa = CALIPH_FOA
	short_name = yes
} # end d_ismaili
